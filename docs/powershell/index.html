User Provisioning Script

# Provision-User.ps1
param($DisplayName, $UserPrincipalName, $Password)
New-ADUser -Name $DisplayName -SamAccountName $UserPrincipalName -AccountPassword (ConvertTo-SecureString $Password -AsPlainText -Force) -Enabled $true
